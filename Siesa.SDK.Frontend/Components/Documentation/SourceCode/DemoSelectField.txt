@using Siesa.SDK.Frontend.Components.Fields
@using Siesa.SDK.Frontend.Components.Visualization
@using Siesa.SDK.Components.Visualization
@using Siesa.Global.Enums

@inject SDKNotificationService Notification


@layout EmptyLayout

<div class="container-fluid">
    <div class="container-fluid">
            <h2>SDKSelectField</h2> 
            <p>Demonstration and configuration of the SDKSelectField component</p> 
    </div>
    <div class="row px-3"> 
        <div class="col-md-6 p-3">
            <SDKCard>
                <h3>SelectField bind-Value</h3>
            <SDKSelectField ItemType="int" @bind-Value="@myValue" Options="@TestList" TextProperty="DisplayText" ValueProperty="Type"></SDKSelectField>
            <p>Item Selected: @myValue</p>
            </SDKCard>
        </div>
        <div class="col-md-6 p-3">
            <SDKCard>
                <h3>SelectField Placeholder and Read    Only</h3>
            <SDKSelectField ItemType="int" @bind-Value="@myValue2" Options="@TestList" TextProperty="DisplayText" 
                            ValueProperty="Type" Placeholder="Hola Mundo" ReadOnly="true"></SDKSelectField>
            <p>Item Selected: @myValue2</p>
            </SDKCard>
        </div>
        <div class="col-md-6 p-3">
            <SDKCard>
                <h3>SelectField ValueChanged</h3>
            <SDKSelectField ItemType="int" Value="@myValue3" Options="@TestList" 
            ValueChanged="(myValue3)=>Onchange(myValue3)" ValueProperty="Type" TextProperty="DisplayText"></SDKSelectField>
            <p>Item Selected: @myValue3</p>
            </SDKCard>
        </div>
         <div class="col-md-6 p-3">
            <SDKCard>
                <h3>SelectField ItemType Enum</h3>
            <SDKSelectField ItemType="enumStatusBaseMaster" @bind-Value="@myValue4" TextProperty="DisplayText" ValueProperty="Type"></SDKSelectField>
            <p>Item Selected: @myValue4</p>
            </SDKCard>
        </div>
    </div>
</div>

@code
{
    
    int myValue;
    
    int myValue2 = 2;

    int myValue3;

    enumStatusBaseMaster myValue4 ;


    List<SDKEnumWrapper<int>> TestList = new List<SDKEnumWrapper<int>>()
    {
        new SDKEnumWrapper<int>() {Type=1, DisplayText="Item 1"},
        new SDKEnumWrapper<int>() {Type=2, DisplayText="Item 2"},
        new SDKEnumWrapper<int>() {Type=3, DisplayText="Item 3"}
    };
    public int Onchange(int item){
    myValue3 = item;
        switch (item)
        {
            case 1:
                Notification.ShowSuccess($"Item Selected {myValue3}");
                return myValue3;
            case 2:
                Notification.ShowWarning($"Item Selected {myValue3}");
                return myValue3;
            case 3:
                Notification.ShowError($"Item Selected {myValue3}");
                return myValue3;
            default:
                return 0;
        }
    }
}
